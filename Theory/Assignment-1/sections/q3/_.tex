    \end{bmatrix}

        x_2 - 2 x_1^2    \\
        x_3 + 3 x_1 x_2  \\

    \end{bmatrix}
    \\
    y & = x_1^2 + x_2
\end{align*}

    \item Find the linearized system around all the equilibrium points.
\end{enumerate}
