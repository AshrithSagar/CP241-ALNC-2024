    \item \begin{align*}
               & \dot{x_{1}}=x_{1}+x_{2}               \\

          \end{align*}
          \vspace*{-1.5em}

    \item \begin{align*}
              \dot{x_{1}} & =x_{1}+x_{2}                 \\
              \dot{x_{2}} & =x_{1} x_{2}^{2}+x_{1}+x_{3} \\

          \end{align*}
\end{enumerate}
